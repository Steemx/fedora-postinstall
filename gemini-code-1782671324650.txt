#!/bin/bash

# Colores para la terminal
VERDE='\033[0;32m'
ANUNCIAR='\033[1;34m'
ROJO='\033[0;31m'
NC='\033[0m' # Sin color

log_status() {
    if [ $1 -eq 0 ]; then
        echo -e "${VERDE}[OK] $2${NC}"
    else
        echo -e "${ROJO}[ERROR] $2${NC}"
        exit 1
    fi
}

# Verificar que se ejecute como root
if [ "$EUID" -ne 0 ]; then
    echo -e "${ROJO}Por favor, ejecuta este script como root (sudo).${NC}"
    exit 1
fi

echo -e "${ANUNCIAR}=== 1. Optimizando DNF5 para Máxima Velocidad ===${NC}"
cat << 'EOF' > /etc/dnf/dnf.conf
[main]
gpgcheck=True
installonly_limit=3
clean_requirements_on_remove=True
best=False
skip_if_unavailable=True
max_parallel_downloads=10
fastestmirror=True
EOF
log_status $? "Configuración de DNF5"

echo -e "${ANUNCIAR}=== 2. Actualizando el Sistema Base ===${NC}"
/usr/bin/dnf upgrade -y
log_status $? "Actualización del sistema"

echo -e "${ANUNCIAR}=== 3. Habilitando Repositorios Clave y Códecs ===${NC}"
# Volvemos a tu método original e infalible de instalación por RPM directo
/usr/bin/dnf install -y https://mirrors.rpmfusion.org/free/fedora/rpmfusion-free-release-$(rpm -E %fedora).noarch.rpm https://mirrors.rpmfusion.org/nonfree/fedora/rpmfusion-nonfree-release-$(rpm -E %fedora).noarch.rpm
/usr/bin/dnf upgrade -y --refresh
/usr/bin/dnf group install "Multimedia" -y
log_status $? "Repositorios RPM Fusion y Códecs"

echo -e "${ANUNCIAR}=== 4. Configurando zRAM al 100% (Optimizado) ===${NC}"
cat << 'EOF' > /etc/systemd/zram-generator.conf
[zram0]
zram-size = ram
compression-algorithm = zstd
EOF
systemctl daemon-reload
systemctl restart systemd-zram-setup@zram0
log_status $? "Configuración de zRAM"

echo -e "${ANUNCIAR}=== 5. Instalando Utilidades del Sistema ===${NC}"
/usr/bin/dnf install -y p7zip p7zip-plugins unrar btop tuned lm_sensors micro
tuned-adm profile balanced
log_status $? "Instalación de utilidades"

echo -e "${ANUNCIAR}=== 6. Instalando Niri, Noctalia y Entorno Base ===${NC}"
/usr/bin/dnf copr enable yalter/niri-git -y
/usr/bin/dnf copr enable lionheartp/Hyprland -y
echo "priority=1" | tee -a /etc/yum.repos.d/_copr:copr.fedorainfracloud.org:yalter:niri-git.repo

# Instalación minimalista acordada (Cero Display Managers)
/usr/bin/dnf install -y niri noctalia-git

/usr/bin/dnf install -y \
    qt6-qtwayland wayland-protocols \
    xdg-desktop-portal-gnome xdg-desktop-portal-gtk \
    pipewire pipewire-pulse wireplumber \
    kitty thunar \
    fira-code-fonts google-noto-sans-fonts \
    kernel-tools gamemode
log_status $? "Instalación del entorno gráfico base"

echo -e "${ANUNCIAR}=== 7. Configurando Auto-Login Nativo en TTY1 ===${NC}"
LOGGED_USER=$(logname || echo $USER)

mkdir -p /etc/systemd/system/getty@tty1.service.d
cat << EOF > /etc/systemd/system/getty@tty1.service.d/override.conf
[Service]
ExecStart=
ExecStart=-/sbin/agetty -o '-p -- \\\\u' --noclear --autologin $LOGGED_USER %I \$TERM
EOF
log_status $? "Configuración de Auto-Login en TTY1"

echo -e "${ANUNCIAR}=== 8. Configurando Arranque Automático de Niri ===${NC}"
USER_HOME=$(eval echo ~$LOGGED_USER)

sed -i '/# Arranque automático de Niri/,+4d' "$USER_HOME/.bash_profile" 2>/dev/null || true

cat << 'EOF' >> "$USER_HOME/.bash_profile"

# Arranque automático de Niri nativo desde TTY1
if [ -z "${DISPLAY}" ] && [ "${XDG_VTNR}" -eq 1 ]; then
    exec niri-session
fi
EOF
chown $LOGGED_USER:$LOGGED_USER "$USER_HOME/.bash_profile"
log_status $? "Configuración de auto-arranque en .bash_profile"

echo -e "${ANUNCIAR}=== 9. Asegurando Desactivación de Gestores Gráficos ===${NC}"
systemctl set-default graphical.target
systemctl disable gdm.service 2>/dev/null || true
systemctl disable lightdm.service 2>/dev/null || true
systemctl disable sddm.service 2>/dev/null || true
systemctl disable greetd.service 2>/dev/null || true
log_status $? "Desactivación de Display Managers anteriores"

echo -e "${ANUNCIAR}=== 10. Habilitando GuC/HuC para Gráficos Intel ===${NC}"
cat << 'EOF' > /etc/modprobe.d/i915.conf
options i915 enable_guc=2
options i915 enable_fbc=1
options i915 modeset=1
EOF
log_status $? "Optimizaciones de video Intel"

echo -e "${VERDE}====================================================${NC}"
echo -e "${VERDE}  ¡Instalación base completada con éxito!            ${NC}"
echo -e "${VERDE}  El sistema se reiniciará en 5 segundos...         ${NC}"
echo -e "${VERDE}====================================================${NC}"
sleep 5
reboot
